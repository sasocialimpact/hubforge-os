    55→export function normalizeConfig(config?: ProviderConfig): ProviderConfig {
    56→  const provider = config?.provider ?? 'zai'
    57→  if (provider === 'zai') return { provider }
    58→  const baseUrl = (config?.baseUrl?.trim() || DEFAULT_BASE_URLS[provider])
    59→  const model = (config?.model?.trim() || DEFAULT_MODELS[provider])
    60→  const apiKey = config?.apiKey?.trim() || ''
    61→  return { provider, baseUrl, model, apiKey }
    62→}

    64→export function describeProvider(config?: ProviderConfig): string {
    65→  const c = normalizeConfig(config)
    66→  if (c.provider === 'zai') return 'Z.ai (shared, free)'
    67→  if (c.provider === 'zai-key') return `Z.ai (own key) · ${c.model}`
    68→  return `${PROVIDER_LABELS[c.provider]} · ${c.model} @ ${c.baseUrl}`
    69→}

    77→// The single LLM entry point. Routes to Z.ai SDK or an OpenAI-compatible fetch.
    78→//
    79→// Rate limiting + retry:
    80→//   - Shared 'zai' provider: wrapped with withSharedSlot (sliding-window queue)
    81→//     + withRetry (exponential backoff on 429/5xx/network errors). This protects
    82→//     the shared key from being overwhelmed when many users hit the platform.
    83→//   - User-key providers ('zai-key', 'openai', etc.): rate-limited by the
    84→//     user's own provider — no queue needed. We still retry on transient errors.
    85→//   - Per-user daily limits are enforced at the API route layer
    86→//     (see src/lib/server/rate-limit-server.ts).
    87→export async function llm(config: ProviderConfig, systemPrompt: string, userPrompt: string): Promise<string> {
    88→  const c = normalizeConfig(config)
    89→  if (c.provider === 'zai') {
    90→    // Shared key — wrap with rate-limit queue + retry.
    91→    const { withSharedSlot, withRetry } = await import('./server/llm-rate-limit')
    92→    return withSharedSlot(() => withRetry(async () => {
    93→      const zai = await getZAI()
    94→      const completion = await zai.chat.completions.create({
    95→        messages: [
    96→          { role: 'assistant', content: systemPrompt },
    97→          { role: 'user', content: userPrompt },
    98→        ],
    99→        thinking: { type: 'disabled' },
   100→      })
   101→      return completion.choices[0]?.message?.content ?? ''
   102→    }))
   103→  }
   104→  // User's own key — retry only, no shared queue.
   105→  const { withRetry } = await import('./server/llm-rate-limit')
   106→  return withRetry(async () => {
   107→    // OpenAI-compatible endpoint
   108→    const url = `${c.baseUrl}/chat/completions`
   109→    const headers: Record<string, string> = { 'Content-Type': 'application/json' }
   110→    if (c.apiKey) headers['Authorization'] = `Bearer ${c.apiKey}`
   111→    const body = {
   112→      model: c.model,
   113→      messages: [
   114→        { role: 'system', content: systemPrompt },
   115→        { role: 'user', content: userPrompt },
   116→      ],
   117→      temperature: 0.7,
   118→    }
   119→    const res = await fetch(url, {
   120→      method: 'POST',
   121→      headers,
   122→      body: JSON.stringify(body),
   123→    })
   124→    if (!res.ok) {
   125→      const txt = await res.text().catch(() => '')
   126→      throw new Error(`Provider ${c.provider} returned ${res.status}: ${txt.slice(0, 300)}`)
   127→    }
   128→    const data = await res.json()
   129→    return data?.choices?.[0]?.message?.content ?? ''
   130→  })
   131→}

   133→// Try to extract a JSON object from an LLM response.
   134→export function extractJSON<T = any>(text: string): T | null {
   135→  if (!text) return null
   136→  let t = text.replace(/```json\s*/gi, '').replace(/```\s*/g, '').trim()
   137→  const objStart = t.indexOf('{')
   138→  const arrStart = t.indexOf('[')
   139→  let start = -1
   140→  let openChar = ''
   141→  let closeChar = ''
   142→  if (objStart === -1 && arrStart === -1) return null
   143→  if (objStart === -1) { start = arrStart; openChar = '['; closeChar = ']' }
   144→  else if (arrStart === -1) { start = objStart; openChar = '{'; closeChar = '}' }
   145→  else {
   146→    start = Math.min(objStart, arrStart)
   147→    openChar = start === objStart ? '{' : '['
   148→    closeChar = start === objStart ? '}' : ']'
   149→  }
   150→  let depth = 0
   151→  let inStr = false
   152→  let escape = false
   153→  for (let i = start; i < t.length; i++) {
   154→    const ch = t[i]
   155→    if (escape) { escape = false; continue }
   156→    if (ch === '\\' && inStr) { escape = true; continue }
   157→    if (ch === '"') { inStr = !inStr; continue }
   158→    if (inStr) continue
   159→    if (ch === openChar) depth++
   160→    else if (ch === closeChar) {
   161→      depth--
   162→      if (depth === 0) {
   163→        const slice = t.slice(start, i + 1)
   164→        try { return JSON.parse(slice) as T } catch { return null }
   165→      }
   166→    }
   167→  }
   168→  return null
   169→}

   195→export interface Decomposition {
   196→  problemStatement: string
   197→  objectives: string[]

   205→export async function supervisorEngine(
   206→  config: ProviderConfig,
   207→  problem: string,
   208→  pack: DomainPack,
   209→  answers?: Record<string, string>
   210→): Promise<Decomposition> {
   211→  const frameworkList = pack.frameworks.map((f) => `- ${f.name}: ${f.description}`).join('\n')
   212→  const answersBlock = answers && Object.keys(answers).length > 0
   213→    ? `\n\nThe user previously answered these clarifying questions - incorporate their answers:\n${Object.entries(answers).map(([id, a]) => `- ${id}: ${a}`).join('\n')}`
   214→    : ''
   215→
   216→  const system = `You are the SUPERVISOR ENGINE of HubForge OS, a recursive reasoning operating system for the ${pack.name}.
   217→Your job: (1) decompose the user's problem into a structured brief, and (2) identify what critical information is MISSING by asking clarifying questions.
   218→
   219→Asking good questions is essential. A program officer may not be an M&E expert. Ask only what truly changes the output - 2 to 4 questions. For each, provide a sensible default assumption the system can use if the user skips it.
   220→
   221→Respond with VALID JSON ONLY. No prose, no markdown fences. Shape:
   222→{
   223→  "problemStatement": "concise restatement",
   224→  "objectives": ["..."],
   225→  "scope": "what is in/out of scope",
   226→  "stakeholders": [{"role": "...", "description": "..."}],
   227→  "keyConsiderations": ["..."],
   228→  "suggestedFrameworks": ["Framework Name 1", ...],
   229→  "clarifyingQuestions": [
   230→    {"id": "q1", "question": "What is...", "why": "We need this because...", "defaultAssumption": "If skipped, we will assume..."}
   231→  ]
   232→}
   233→
   234→Available frameworks in the ${pack.name}:
   235→${frameworkList}`
   236→
   237→  const user = `Decompose this problem for the ${pack.domain} domain:${answersBlock}
   238→
   239→${problem}`
   240→
   241→  const raw = await llm(config, system, user)
   242→  const parsed = extractJSON<Decomposition>(raw)
   243→  if (parsed) {
   244→    if (!parsed.clarifyingQuestions) parsed.clarifyingQuestions = []
   245→    return parsed
   246→  }
   247→  return {
   248→    problemStatement: problem,
   249→    objectives: ['Produce an expert-grade response to the stated problem.'],
   250→    scope: 'As stated by the user.',
   251→    stakeholders: [],
   252→    keyConsiderations: [],
   253→    suggestedFrameworks: pack.frameworks.slice(0, 2).map((f) => f.name),
   254→    clarifyingQuestions: [],
   255→  }
   256→}

   261→export interface RetrievalResult {
   262→  frameworks: DomainPack['frameworks']
   263→  decisionRules: DomainPack['decisionRules']
   264→  evidence: DomainPack['evidence']
   265→  historicalMemory: DomainPack['historicalMemory']
   266→  reasoningPatterns: DomainPack['reasoningPatterns']
   267→  improvementHeuristics: DomainPack['improvementHeuristics']
   268→  procedures: DomainPack['procedures']
   269→}
   270→
   271→export function retrievalEngine(
   272→  _problem: string,
   273→  decomposition: Decomposition,
   274→  pack: DomainPack
   275→): RetrievalResult {
   276→  const suggested = new Set((decomposition.suggestedFrameworks || []).map((n) => n.toLowerCase()))
   277→  const frameworks = pack.frameworks.filter((f) => suggested.has(f.name.toLowerCase()))
   278→  const finalFrameworks = frameworks.length > 0 ? frameworks : pack.frameworks.slice(0, 3)
   279→  return {
   280→    frameworks: finalFrameworks,
   281→    decisionRules: pack.decisionRules,
   282→    evidence: pack.evidence,
   283→    historicalMemory: pack.historicalMemory,
   284→    reasoningPatterns: pack.reasoningPatterns,
   285→    improvementHeuristics: pack.improvementHeuristics,
   286→    procedures: pack.procedures,
   287→  }
   288→}

   290→// ============================================================
   291→// Engine 3 - Rule Engine (deterministic, no LLM)
   292→// ============================================================
   293→export interface RuleCheckResult {
   294→  rule: string
   295→  passed: boolean
   296→  note: string
   297→}
   298→
   299→export function ruleEngine(problem: string, pack: DomainPack): RuleCheckResult[] {
   300→  const p = problem.toLowerCase()
   301→  const results: RuleCheckResult[] = []
   302→  for (const rule of pack.decisionRules) {
   303→    let passed = false
   304→    let note = ''
   305→    switch (rule.name) {
   306→      case 'SMART Goal Validation':
   307→        passed = /\b(by|within|target|increase|reduce|%\d|%\s)/i.test(problem)
   308→        note = passed ? 'Problem references a measurable target or time horizon.' : 'No explicit measurable target detected - Reasoning Engine must add SMART targets.'
   309→        break
   310→      case 'Stakeholder Coverage':
   311→        passed = /\b(farmer|beneficiar|communit|patient|student|women|youth|household|government|partner)/i.test(p)
   312→        note = passed ? 'At least one stakeholder group named.' : 'No stakeholder group explicitly named - Reasoning Engine must enumerate.'
   313→        break
   314→      case 'Assumption Explicitness':
   315→        passed = /\b(assum|given that|provided that|if .+ holds)/i.test(p)
   316→        note = passed ? 'Problem hints at assumptions.' : 'No assumptions stated - Critique Engine will demand explicit assumptions.'
   317→        break
   318→      case 'Evidence Citation':
   319→        passed = false
   320→        note = 'User problem cites no evidence - Reasoning Engine must ground claims in the Evidence Library.'
   321→        break
   322→      case 'Risk Identification':
   323→        passed = /\b(risk|threat|shock|uncertain|hazard)/i.test(p)
   324→        note = passed ? 'Problem mentions risk explicitly.' : 'No risk language detected - Critique Engine will require risk analysis.'
   325→        break
   326→      default:
   327→        passed = true
   328→        note = 'Rule checked.'
   329→    }
   330→    results.push({ rule: rule.name, passed, note })
   331→  }
   332→  return results
   333→}

   338→export async function reasoningEngine(
   339→  config: ProviderConfig,
   340→  problem: string,
   341→  decomposition: Decomposition,
   342→  retrieval: RetrievalResult,
   343→  priorCritique: string | null,
   344→  priorDraft: string | null,
   345→  pack: DomainPack,
   346→  iteration: number,
   347→  maxIterations: number,
   348→  outputTypes: OutputType[],
   349→  answers?: Record<string, string>,
   350→  webSearch?: { demographic: any[]; previousPrograms: any[]; evidence: any[]; summary: string } | null,
   351→  orgContext?: string | null,
   352→  contextBlocks?: string | null,
   353→): Promise<string> {
   354→  const frameworksText = (retrieval.frameworks || [])
   355→    .map((f: any) => `### ${f.name}\n${f.description || ''}\nWhen to use: ${f.whenToUse || ''}\nKey elements: ${(f.keyElements || []).join(', ')}${f.template ? `\nTemplate: ${f.template}` : ''}`)
   356→    .join('\n\n')
   357→  const rulesText = (retrieval.decisionRules || []).map((r: any) => `- ${r.name}: ${r.check || ''} (pass: ${r.passCondition || ''}; fail: ${r.failAction || ''})`).join('\n')
   358→  const evidenceText = (retrieval.evidence || []).map((e: any) => `- ${e.title} (${e.type}): ${e.summary || ''}`).join('\n')
   359→  const memoryText = (retrieval.historicalMemory || []).map((m: any) => `- Problem: ${m.problem || ''}\n  Context: ${m.context || ''}\n  Outcome: ${m.outcome || ''}\n  Lesson: ${m.lesson || ''}`).join('\n')
   360→  const patternsText = (retrieval.reasoningPatterns || []).map((p: any) => `- ${p.name}: ${p.description || ''}`).join('\n')
   361→  const answersBlock = answers && Object.keys(answers).length > 0
   362→    ? `\n## User's answers to clarifying questions (treat as authoritative input)\n${Object.entries(answers).map(([id, a]) => `- ${id}: ${a}`).join('\n')}`
   363→    : ''
   364→  const outputGuidance = outputTypes.length > 0
   365→    ? `\n## Requested deliverables\nThe user wants: ${outputTypes.map((o) => OUTPUT_LABELS[o].label).join(', ')}. Produce a single unified Markdown strategy document that serves as the source for all requested deliverables. Use clear sections (##) for each logical part. If a Theory of Change is requested, include a section titled "## Theory of Change" with bullet lists for Inputs, Activities, Outputs, Outcomes, Impact, Assumptions, and External Factors. If a Logframe is requested, include a section "## Logframe" with a Markdown table.`
   366→    : ''
   367→
   368→  // Web search context (demographic data, previous programs, evidence from the live web)
   369→  const webSearchBlock = webSearch && (webSearch.summary || webSearch.demographic?.length || webSearch.previousPrograms?.length || webSearch.evidence?.length)
   370→    ? `\n## Live Web Research (demographics, previous programs, evidence)
   371→
   372→### Research Summary
   373→${webSearch.summary || 'No summary available.'}
   374→
   375→### Demographic Data & Context
   376→${(webSearch.demographic || []).slice(0, 5).map((r: any, i: number) => `${i + 1}. ${r.title || ''}\n   ${r.snippet || ''}\n   Source: ${r.source || r.url || ''}`).join('\n') || 'No demographic data found.'}
   377→
   378→### Previous Programs & Initiatives
   379→${(webSearch.previousPrograms || []).slice(0, 5).map((r: any, i: number) => `${i + 1}. ${r.title || ''}\n   ${r.snippet || ''}\n   Source: ${r.source || r.url || ''}`).join('\n') || 'No previous programs found.'}
   380→
   381→### Evidence & Research
   382→${(webSearch.evidence || []).slice(0, 5).map((r: any, i: number) => `${i + 1}. ${r.title || ''}\n   ${r.snippet || ''}\n   Source: ${r.source || r.url || ''}`).join('\n') || 'No evidence found.'}
   383→
   384→IMPORTANT: Use this live web research to ground your strategy in real demographic data, learn from previous programs in the target region, and cite evidence. Reference specific data points and programs by name.`
   385→    : ''
   386→
   387→  const system = `You are the REASONING ENGINE of HubForge OS, operating with the ${pack.name}.
   388→Your task is to produce an expert-grade draft response to the user's problem using the retrieved knowledge.
   389→
   390→Iteration: ${iteration} of ${maxIterations}.
   391→
   392→REQUIREMENTS:
   393→1. Ground every empirical claim in the Evidence Library or Historical Memory. Cite sources inline like [Source: <title>].
   394→2. Apply the retrieved Frameworks explicitly. Name them as you use them.
   395→3. Satisfy every Decision Rule. If a rule's pass condition requires a measurable target, state one.
   396→4. Use the Reasoning Patterns to structure your analysis.
   397→5. Be specific. Replace vague outputs with measurable targets.
   398→6. Output in well-structured Markdown with clear ## sections.`
   399→
   400→  const user = `# PROBLEM
   401→${problem}${answersBlock}
   402→${outputGuidance}
   403→${orgContext || ''}
   404→${contextBlocks || ''}
   405→${webSearchBlock}
   406→
   407→# DECOMPOSITION (from Supervisor Engine)
   408→- Problem statement: ${decomposition.problemStatement || ''}
   409→- Objectives: ${(decomposition.objectives || []).join('; ')}
   410→- Scope: ${decomposition.scope || ''}
   411→- Stakeholders: ${(decomposition.stakeholders || []).map((s: any) => `${s.role} (${s.description || ''})`).join('; ')}
   412→- Key considerations: ${(decomposition.keyConsiderations || []).join('; ')}
   413→
   414→# RETRIEVED KNOWLEDGE
   415→
   416→## Frameworks
   417→${frameworksText}
   418→
   419→## Decision Rules (must satisfy)
   420→${rulesText}
   421→
   422→## Evidence Library
   423→${evidenceText}
   424→
   425→## Historical Memory (prior cases)
   426→${memoryText}
   427→
   428→## Reasoning Patterns
   429→${patternsText}
   430→${priorDraft && priorCritique ? `
   431→
   432→# PRIOR ITERATION (iteration ${iteration - 1})
   433→You produced a draft that was critiqued. You MUST address every critique point.
   434→
   435→## Prior Draft
   436→${priorDraft}
   437→
   438→## Critique to Address
   439→${priorCritique}
   440→` : ''}
   441→
   442→# TASK
   443→Produce the best expert-grade draft response you can. Be specific, evidence-grounded, and structured.`
   444→
   445→  return await llm(config, system, user)
   446→}

   462→export async function critiqueEngine(config: ProviderConfig, draft: string, pack: DomainPack): Promise<CritiqueResult> {
   463→  const heuristicsText = pack.improvementHeuristics.map((h) => `- ${h.name}: ${h.description}`).join('\n')
   464→  const system = `You are the CRITIQUE ENGINE of HubForge OS, operating with the ${pack.name}.
   465→Find weaknesses in the draft using the Improvement Heuristics. Be rigorous and specific. Quote the offending text where possible.
   466→
   467→Respond with VALID JSON ONLY. Shape:
   468→{
   469→  "issues": [{"severity": "high|medium|low", "heuristic": "<name>", "description": "<what is wrong and where>"}],
   470→  "summary": "<one or two sentence summary>"
   471→}
   472→
   473→Improvement Heuristics:
   474→${heuristicsText}`
   475→  const user = `Critique this draft:\n\n${draft}`
   476→  const raw = await llm(config, system, user)
   477→  const parsed = extractJSON<CritiqueResult>(raw)
   478→  if (parsed && Array.isArray(parsed.issues)) return parsed
   479→  return { issues: [{ severity: 'medium', heuristic: 'General', description: 'Critique engine could not produce a structured critique; manual review required.' }], summary: 'Critique parsing failed.' }
   480→}

   487→  const system = `You are the IMPROVEMENT ENGINE of HubForge OS, operating with the ${pack.name}.
   488→You receive a draft and a critique. Produce an IMPROVED draft that fixes every critique issue while preserving strengths. Keep structure clean. Output full improved draft in Markdown.`
   489→  const user = `# DRAFT TO IMPROVE\n${draft}\n\n# CRITIQUE TO ADDRESS (fix every issue)\n${issuesText}\n\nSummary: ${critique.summary}\n\n# TASK\nProduce the improved draft.`

   493→// ============================================================
   494→// Engine 7 - Evaluation Engine
   495→// ============================================================
   496→export interface EvaluationResult {
   497→  scores: { criterion: string; score: number; weight: number; rationale: string }[]
   498→  overall: number
   499→  thresholdMet: boolean
   500→  notes: string
   501→}
   502→
   503→export async function evaluationEngine(config: ProviderConfig, draft: string, pack: DomainPack, threshold: number): Promise<EvaluationResult> {
   504→  const rubricText = pack.evaluationCriteria.map((c) => `- ${c.criterion} (weight ${c.weight}): ${c.description}\n  Scoring: ${c.scoringGuide}`).join('\n')
   505→  const system = `You are the EVALUATION ENGINE of HubForge OS, operating with the ${pack.name}.
   506→Score the draft against the rubric. Each criterion 0-100. Respond with VALID JSON ONLY. Shape:
   507→{"scores": [{"criterion": "<exact name>", "score": <0-100>, "weight": <weight>, "rationale": "<1 sentence>"}], "overall": <weighted avg>, "notes": "<1-2 sentences>"}
   508→
   509→Rubric:
   510→${rubricText}
   511→
   512→Threshold for delivery: ${threshold}.`
   513→  const user = `Score this draft:\n\n${draft}`
   514→  const raw = await llm(config, system, user)
   515→  const parsed = extractJSON<{ scores: any[]; overall: number; notes: string }>(raw)
   516→  if (parsed && Array.isArray(parsed.scores) && parsed.scores.length > 0) {
   517→    const weightByName = new Map(pack.evaluationCriteria.map((c) => [c.criterion.toLowerCase(), c.weight]))
   518→    const scores = parsed.scores.map((s) => {
   519→      const w = weightByName.get(String(s.criterion).toLowerCase()) ?? s.weight ?? 0
   520→      return { criterion: s.criterion, score: Number(s.score) || 0, weight: w, rationale: String(s.rationale ?? '') }
   521→    })
   522→    const totalW = scores.reduce((a, s) => a + s.weight, 0) || 1
   523→    const overall = Math.round(scores.reduce((a, s) => a + s.score * s.weight, 0) / totalW)
   524→    return { scores, overall, thresholdMet: overall >= threshold, notes: parsed.notes ?? '' }
   525→  }
   526→  return {
   527→    scores: pack.evaluationCriteria.map((c) => ({ criterion: c.criterion, score: 60, weight: c.weight, rationale: 'Evaluation parse failed; neutral score.' })),
   528→    overall: 60, thresholdMet: 60 >= threshold, notes: 'Evaluation parsing failed.',
   529→  }
   530→}

   588→  activities: LogframeRow[]

   591→export interface StructuredOutputs {
   592→  toc?: ToCData
   593→  logframe?: LogframeData
   594→}
   595→
   596→export async function structureEngine(
   597→  config: ProviderConfig,
   598→  finalDraft: string,
   599→  outputTypes: OutputType[]
   600→): Promise<StructuredOutputs> {
   601→  const result: StructuredOutputs = {}
   602→  const wantToc = outputTypes.includes('toc')
   603→  const wantLogframe = outputTypes.includes('logframe')
   604→
   605→  if (wantToc) {
   606→    try {
   607→      const system = `You are the STRUCTURE ENGINE of HubForge OS. Extract a Theory of Change from the strategy document and return VALID JSON ONLY. Shape:
   608→{"targetPopulation": "...", "inputs": ["..."], "activities": ["..."], "outputs": ["..."], "outcomes": ["..."], "impact": "...", "assumptions": ["..."], "externalFactors": ["..."]}
   609→Each list should have 2-6 concise items derived from the document. If a field is not present, infer it from context or use an empty string/array.`
   610→      const user = `Extract the Theory of Change from:\n\n${finalDraft}`
   611→      const raw = await llm(config, system, user)
   612→      const parsed = extractJSON<ToCData>(raw)
   613→      if (parsed) result.toc = {
   614→        targetPopulation: String(parsed.targetPopulation ?? ''),
   615→        inputs: arr(parsed.inputs), activities: arr(parsed.activities), outputs: arr(parsed.outputs),
   616→        outcomes: arr(parsed.outcomes), impact: String(parsed.impact ?? ''),
   617→        assumptions: arr(parsed.assumptions), externalFactors: arr(parsed.externalFactors),
   618→      }
   619→    } catch { /* leave undefined */ }
   620→  }
   621→
   622→  if (wantLogframe) {
   623→    try {
   624→      const system = `You are the STRUCTURE ENGINE of HubForge OS. Extract a Logical Framework from the strategy document and return VALID JSON ONLY. Shape:
   625→{"goal": {"level":"Goal","description":"...","ovi":"...","mov":"...","assumptions":"..."}, "purpose": {...}, "outputs": [{...}], "activities": [{...}]}
   626→Goal and Purpose are single rows; outputs and activities are arrays of 2-5 rows each. Infer fields from the document where needed.`
   627→      const user = `Extract the Logframe from:\n\n${finalDraft}`
   628→      const raw = await llm(config, system, user)
   629→      const parsed = extractJSON<any>(raw)
   630→      if (parsed) {
   631→        result.logframe = {
   632→          goal: row(parsed.goal, 'Goal'),
   633→          purpose: row(parsed.purpose, 'Purpose'),
   634→          outputs: (Array.isArray(parsed.outputs) ? parsed.outputs : []).map((r: any) => row(r, 'Output')),
   635→          activities: (Array.isArray(parsed.activities) ? parsed.activities : []).map((r: any) => row(r, 'Activity')),
   636→        }
   637→      }
   638→    } catch { /* leave undefined */ }
   639→  }
   640→
   641→  return result
   642→}
   643→
   644→function arr(v: any): string[] { return Array.isArray(v) ? v.map((x) => String(x)).filter(Boolean) : [] }
   645→function row(r: any, level: string): LogframeRow {
   646→  return { level, description: String(r?.description ?? ''), ovi: String(r?.ovi ?? ''), mov: String(r?.mov ?? ''), assumptions: String(r?.assumptions ?? '') }
   647→}

   649→// ============================================================
   650→// Feedback Engine - incorporates user feedback into a new draft.
   651→// ============================================================
   652→export async function feedbackEngine(
   653→  config: ProviderConfig,
   654→  currentDraft: string,
   655→  feedback: string,
   656→  pack: DomainPack
   657→): Promise<{ improved: string; feedbackAddressed: string[] }> {
   658→  const system = `You are the FEEDBACK ENGINE of HubForge OS, operating with the ${pack.name}.
   659→The user reviewed the deliverable and gave feedback. Revise the draft to address the feedback. Preserve strengths. Do not introduce unsupported claims. Output the full revised draft in Markdown.
   660→
   661→After the draft, on a new line, output a line starting with "ADDRESSED:" followed by a JSON array of strings describing what you changed, e.g.:
   662→ADDRESSED: ["Made assumptions about market access explicit", "Added a risk row on input prices"]`
   663→  const user = `# CURRENT DRAFT\n${currentDraft}\n\n# USER FEEDBACK\n${feedback}\n\n# TASK\nRevise the draft to address the feedback. Then output the ADDRESSED: line.`
   664→  const raw = await llm(config, system, user)
   665→  // Split draft from ADDRESSED marker
   666→  const idx = raw.indexOf('ADDRESSED:')
   667→  let improved = raw
   668→  let addressed: string[] = []
   669→  if (idx >= 0) {
   670→    improved = raw.slice(0, idx).trim()
   671→    const after = raw.slice(idx + 'ADDRESSED:'.length).trim()
   672→    const parsed = extractJSON<string[]>(after)
   673→    if (Array.isArray(parsed)) addressed = parsed
   674→  }
   675→  return { improved, feedbackAddressed: addressed }
   676→}
